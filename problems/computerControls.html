电脑控制与录屏

Table of Contents for current page

TIP

实现方案：
1.使用noNvc实现电脑控制(被控电脑需要安装客户端，一直下一步，然后配置好IP，端口和密码即可)
2.使用navigator.mediaDevices.getDisplayMedia() API实现录屏(API需要获取浏览器权限，localhost和https环境可以正常运行，在http环境会有浏览器权限限制，直接报错)

直接新建组件(smallWindow.vue)

vue
<template>
    <div class="page-box" :ref="'canvas' + id">
        <div id="screen"></div>
    </div>
</template>
<script>
import RFB from '@novnc/novnc'
export default {
  name: 'smallWindow',
  props: {
    id: {
      type: String,
      default: ''
    },
    // 服务地址
    vncServer: {
      type: String,
      default: ''
    },
    // 可选密码
    password: {
      type: String,
      default: ''
    },
  },
  data() {
    return {
      rfb: null,
      isFullscreen: false
    }
  },
  methods: {
    // 链接nvc
    connectVnc() {
      this.closeVnc()
      console.log(this.vncServer,this.password)
      this.rfb = new RFB(this.$refs['canvas' + this.id].querySelector('#screen'),this.vncServer,{
        credentials: {
          password: this.password,
        }
      })
      this.rfb.addEventListener("connect", this.connectedToServer);
      this.rfb.addEventListener("disconnect", this.disconnectedFromServer);
      this.rfb.scaleViewport = true; // true为自适应大小，false当远程会话小于容器大小时居中
      this.rfb.resizeSession = false;
    },
    closeVnc() {
      if(this.rfb) {
        this.rfb && this.rfb.disconnect();
        this.rfb = null
      }
      const screenElement = this.$refs['canvas' + this.id].querySelector('#screen')
      if (screenElement) {
        screenElement.innerHTML = '';
      }
    },
    connectedToServer(e) {
      console.log("success", e);
    },
    disconnectedFromServer(e) {
      // 根据情况重新连接或销毁
      if (e.detail.clean) {
        // 重新连接
        // this.connectVnc()
        // 销毁连接
        // this.rfb && this.rfb.disconnect();
      } else {
        // 重新连接
        // this.connectVnc()
        // 销毁连接
        // this.rfb && this.rfb.disconnect();
      }
    }
  }
}
</script>

<template>
  <page-layout>
     <template slot="header">
            <a-form layout="inline" :colon="false">

                        <a-button type="primary" @click="handleQuery">查询</a-button>
                        <a-button @click="resetQuery">重置</a-button>
                    </a-space>
                </a-form-item>
            </a-form>
        </template>
    <div class="smallViewport">
        <div class="item" :id="'item' + index" v-for="(item, index) in DataList" :key="index">
            <div class="name">
                <div class="name-ite">
                    {{item.name}}
                </div>
                <div style="width:50%;display: flex;justify-content: flex-end; align-items: center;">
                    <a-button style="background:rgba(0,0,0,0);color: #fff;" type="link" @click="toFullScreen(index)"> 全屏 </a-button>
                    <a-button style="background:rgba(0,0,0,0);color: #fff;" type="link" @click="record(index)"> 录屏 </a-button>
                    <a-button style="background:rgba(0,0,0,0);color: #fff;" type="link" @click="again(item)"> 重连 </a-button>
                </div>
            </div>
            <smallWindow :id="item.id" :ref="'smallWindow' + item.id" :vncServer="'ws://' + item.ip +':' + item.port" :password="decrypt(keyAP,item.password)" />
        </div>
    </div>
    <div class="page-area">
      <a-pagination
         :current= "pageCfg.current"
          :pageSize= "pageCfg.size"
          :total= "Number(total)"
          @change="changeSize"
          @showSizeChange="onShowSizeChange"
          hideOnSinglePage
         show-size-changer 
         show-quick-jumper />
    </div>
  </page-layout>
</template>

<script>
import smallWindow from './components/smallWindow.vue'
import { queryItem, pageConfig } from './config/column.js';
import { getList, deleteList } from './config/api.js' 
import screenfull from "screenfull"; // 元素全屏
import { getKey } from '@/services/password.js' // 获取密码加减密KEY
import { decrypt } from '@/utils/aes.js' // 密码解密方法
export default {
  name: 'computerControl',
  components: {
    smallWindow
  },
  data() {
    return {
      queryParam: { ...queryItem },
            pageCfg: { ...pageConfig },
             total: 100,
            DataList: [],
            keyAP: '',
            decrypt,
    }
  },
  mounted() {
        this.getKey()
        this.getList();
    },
    methods: {
        getKey() {
            getKey().then(res => {
                this.keyAP = res.data
            })
        },
        // 录屏
        async record() {
            const stream = await navigator.mediaDevices.getDisplayMedia({video: true})
            const mime = MediaRecorder.isTypeSupported("video/webm;codecs=h264") ? "video/webm;codecs=h264" : "video/webm"
            const mediaRecorder = new MediaRecorder(stream, {mimeType: mime})
            const chunks = []
            mediaRecorder.addEventListener("dataavailable", e => {
                chunks.push(e.data)
            })
            mediaRecorder.addEventListener("stop", () => {
                const blob = new Blob(chunks,{type: chunks[0].type})
                const url = URL.createObjectURL(blob)
                console.log(url,'url')
                const a = document.createElement('a')
                a.href = url
                a.download = 'video.mp4'
                a.click()
            })
            mediaRecorder.start()
        },
        toFullScreen(index) {
            const el = document.getElementById(`item${index}`)
            if (!el) {
                this.$message.error('未找到要全屏的元素')
                return
            }
            if (screenfull.isEnabled) {
                screenfull.toggle(el)
            } else {
                this.$message.warning('您的浏览器不支持全屏功能')
            }
        },
        handleQuery() { // 查询
            Object.assign(this.pageCfg, { ...pageConfig })
            this.getList()
        },
        resetQuery() { // 重置
            Object.assign(this.queryParam, { ...queryItem })
            Object.assign(this.pageCfg, { ...pageConfig })
            this.roomList1 = []
            this.getList()
        },
        changeSize(current, pageSize) {
            this.pageCfg.current = current
            this.pageCfg.size = pageSize
            this.getList()
        },
        onShowSizeChange(current, pageSize) {
            this.pageCfg.current = 1
            this.pageCfg.size = pageSize
            this.getList()
        },
        getList() {
            this.loading = true;
            getList({
                ...this.queryParam,
                ...this.pageCfg
            }).then((res) => {
                if (res.code == 0) {
                    this.DataList = res.data.records;
                    this.total = res.data.total;
                    this.$nextTick(() => {
                        this.DataList.forEach(item => {
                            this.$refs['smallWindow' + item.id][0].connectVnc()
                        })
                    })
                }
                this.loading = false;
            })
            .catch((err) => {
                this.loading = false;
            });

            this.loading = false;
        },
        // 重新连接
        again(item) {
            this.$refs['smallWindow' + item.id][0].connectVnc()
        },
    }
}
</script>
